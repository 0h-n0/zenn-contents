#### Gateway基盤の実装例

以下はPythonでのGateway骨格です。認証・ルーティング・キャッシュの3層を持ちます。HTTP/2によるコネクション多重化を使うため、`httpx[http2]`が必要です（`pip install httpx[http2]`）。

```python
# gateway.py
import time
from dataclasses import dataclass, field
from typing import Any
import httpx

@dataclass
class MCPServerConfig:
    name: str
    url: str
    tools: list[str]
    allowed_roles: dict[str, list[str]] = field(default_factory=dict)

class MCPGateway:
    def __init__(self, servers: list[MCPServerConfig]) -> None:
        self._servers = {s.name: s for s in servers}
        self._tool_to_server: dict[str, str] = {}
        self._cache: dict[str, tuple[Any, float]] = {}
        self._client = httpx.AsyncClient(
            limits=httpx.Limits(max_connections=100, max_keepalive_connections=20),
            timeout=httpx.Timeout(10.0, connect=2.0),
            http2=True,  # コネクション多重化
        )
        for s in servers:
            for t in s.tools:
                self._tool_to_server[t] = s.name

    async def call_tool(
        self, tool_name: str, args: dict, user_role: str
    ) -> dict:
        server = self._servers[self._tool_to_server[tool_name]]
        # RBAC チェック
        if roles := server.allowed_roles.get(tool_name):
            if user_role not in roles:
                raise PermissionError(f"{user_role} cannot access {tool_name}")
        # キャッシュ確認
        key = f"{tool_name}:{hash(frozenset(args.items()))}"
        if key in self._cache:
            val, ts = self._cache[key]
            if time.monotonic() - ts < 300:
                return val
        # JSON-RPC転送
        resp = await self._client.post(f"{server.url}/mcp", json={
            "jsonrpc": "2.0", "method": "tools/call",
            "params": {"name": tool_name, "arguments": args},
        })
        result = resp.json().get("result", {})
        self._cache[key] = (result, time.monotonic())
        return result
```

**設計判断**: `httpx.Limits`で接続プールを管理し、`http2=True`でコネクション多重化を有効化しています。最初は`max_connections`を大きくしがちですが、ダウンストリームの過負荷を防ぐため現実的なピーク負荷に合わせるべきです。

#### バッチングで往復を削減する

独立したツール呼び出しを並列実行し、逐次5,500ms（1,100ms×5）を約1,200-1,500msに短縮できます。

```python
import asyncio

async def batch_tool_calls(
    gw: MCPGateway, calls: list[dict], user_role: str, max_concurrency: int = 5
) -> list[dict]:
    sem = asyncio.Semaphore(max_concurrency)
    async def _call(c: dict) -> dict:
        async with sem:
            return await gw.call_tool(c["tool"], c["args"], user_role)
    results = await asyncio.gather(*[_call(c) for c in calls], return_exceptions=True)
    return [r if not isinstance(r, Exception) else {"error": str(r)} for r in results]
```

**トレードオフ**: 並列度を上げすぎるとダウンストリームが過負荷になります。`max_concurrency`で制御してください。

## OAuth 2.0 + RBACで最小権限のツール公開を実現する

MCP Gatewayの認証設計では、**OAuth 2.0による認証**と**RBAC（Role-Based Access Control）による認可**を組み合わせるパターンが推奨されます。MCP仕様の[Security and Trust & Safety](https://modelcontextprotocol.io/specification/2025-11-25)セクションでは、「ユーザーは全てのデータアクセスと操作に明示的に同意する必要がある」と定められています。

### 認証フロー: OAuth 2.0 + JWTトークン

エンタープライズ環境では、既存のIdP（Identity Provider）と連携するOAuth 2.0フローが一般的です。以下はGatewayでの認証ミドルウェア実装例です。

```python
# auth.py
from dataclasses import dataclass
from datetime import datetime, timezone
import json
import base64
from typing import Any

@dataclass
class TokenClaims:
    sub: str         # ユーザーID
    roles: list[str] # 付与されたロール
    exp: float       # 有効期限 (Unix timestamp)
    iss: str         # 発行者

class AuthMiddleware:
    """OAuth 2.0 Bearer Token を検証し、RBACを適用するミドルウェア"""

    def __init__(self, issuer: str, allowed_issuers: list[str] | None = None) -> None:
        self._issuer = issuer
        self._allowed_issuers = allowed_issuers or [issuer]
        self._revoked_tokens: set[str] = set()  # トークン失効リスト

    def decode_token(self, token: str) -> TokenClaims:
        """JWTトークンのペイロードをデコード（署名検証は省略）"""
        # 本番では PyJWT + 公開鍵検証を使用すること
        parts = token.split(".")
        if len(parts) != 3:
            raise ValueError("Invalid JWT format")
        payload = json.loads(
            base64.urlsafe_b64decode(parts[1] + "==").decode()
        )
        return TokenClaims(
            sub=payload["sub"],
            roles=payload.get("roles", []),
            exp=payload["exp"],
            iss=payload["iss"],
        )

    def validate_token(self, token: str) -> TokenClaims:
        """トークンの有効性を検証"""
        if token in self._revoked_tokens:
            raise PermissionError("Token has been revoked")
        claims = self.decode_token(token)
        # 有効期限チェック
        if claims.exp < datetime.now(timezone.utc).timestamp():
            raise PermissionError("Token expired")
        # 発行者チェック
        if claims.iss not in self._allowed_issuers:
            raise PermissionError(f"Untrusted issuer: {claims.iss}")
        return claims

    def check_permission(
        self, claims: TokenClaims, tool_name: str,
        tool_roles: dict[str, list[str]]
    ) -> bool:
        """ツール単位のRBACチェック"""
        required_roles = tool_roles.get(tool_name)
        if required_roles is None:
            return True  # ロール制限なし
        return any(r in required_roles for r in claims.roles)
```

**設計判断**: トークンの署名検証は本番環境では必ずPyJWT + RSA/ECDSA公開鍵で行ってください。上記のデコード処理は構造説明のための簡略版です。

### RBACポリシーの設計指針

ツール公開時のロール設計では、**最小権限の原則**に基づいて以下のように分離します。

| ロール | 許可されるツール | ユースケース |
|--------|----------------|-------------|
| `viewer` | `search`, `get_status` | 情報閲覧のみ |
| `editor` | `viewer` + `create_ticket`, `update_doc` | データ変更を含む操作 |
| `admin` | `editor` + `delete_*`, `manage_users` | 管理者操作 |
| `service` | 特定ツールのみ | サービス間連携（M2M） |

```python
# rbac_config.py
TOOL_PERMISSIONS: dict[str, list[str]] = {
    # 閲覧系: 全ロールに開放
    "slack_search": ["viewer", "editor", "admin"],
    "jira_get_issue": ["viewer", "editor", "admin"],
    "db_query": ["viewer", "editor", "admin"],
    # 変更系: editor以上
    "jira_create_issue": ["editor", "admin"],
    "slack_post_message": ["editor", "admin"],
    "db_insert": ["editor", "admin"],
    # 管理系: adminのみ
    "jira_delete_issue": ["admin"],
    "slack_manage_channel": ["admin"],
    "db_drop_table": ["admin"],
    # M2M: serviceロール専用
    "sync_data": ["service"],
}
```

**注意点**: LLMエージェントがツールを呼び出す場合、エージェントではなく**リクエストを開始したユーザーの権限**で認可を判定してください。エージェントに高い権限を付与すると、プロンプトインジェクションにより意図しない操作が実行されるリスクがあります。

### プロンプトインジェクション対策

MCP環境特有のリスクとして、LLMがユーザー入力をツール呼び出しに変換する過程で、悪意あるプロンプトが混入する可能性があります。

```python
# injection_guard.py
import re

# 危険なパターンの検出
SUSPICIOUS_PATTERNS = [
    r"(?i)ignore\s+previous\s+instructions",
    r"(?i)system\s*:\s*you\s+are",
    r"(?i)forget\s+(everything|all|your)",
    r"(?i)admin\s+override",
]

def sanitize_tool_args(args: dict[str, str]) -> dict[str, str]:
    """ツール引数からプロンプトインジェクションの疑いがあるパターンを検出"""
    sanitized = {}
    for key, value in args.items():
        if isinstance(value, str):
            for pattern in SUSPICIOUS_PATTERNS:
                if re.search(pattern, value):
                    raise ValueError(
                        f"Suspicious input detected in '{key}': "
                        f"possible prompt injection"
                    )
        sanitized[key] = value
    return sanitized
```

**制約条件**: 正規表現ベースのフィルタリングは完全ではなく、巧妙な攻撃を見逃す可能性があります。多層防御として、ツール側でも入力バリデーションを実施し、破壊的操作には人間の承認フローを追加することを推奨します。

### 監査ログの設計

セキュリティ要件として、ツール呼び出しの監査ログは必須です。誰が・いつ・何を・どの結果で実行したかを記録します。

```python
# audit.py
import json
import time
from dataclasses import dataclass, asdict

@dataclass
class AuditEntry:
    timestamp: float
    user_id: str
    user_role: str
    tool_name: str
    args_hash: str  # 引数のハッシュ（PII保護のため原文は記録しない）
    result_status: str  # "success" | "denied" | "error"
    latency_ms: float

    def to_json(self) -> str:
        return json.dumps(asdict(self), ensure_ascii=False)

class AuditLogger:
    def __init__(self, log_path: str = "/var/log/mcp-gateway/audit.jsonl") -> None:
        self._log_path = log_path

    def log(self, entry: AuditEntry) -> None:
        with open(self._log_path, "a") as f:
            f.write(entry.to_json() + "\n")

    def log_tool_call(
        self, user_id: str, role: str, tool: str,
        args_hash: str, status: str, latency: float
    ) -> None:
        entry = AuditEntry(
            timestamp=time.time(),
            user_id=user_id,
            user_role=role,
            tool_name=tool,
            args_hash=args_hash,
            result_status=status,
            latency_ms=latency,
        )
        self.log(entry)
```

**トレードオフ**: ツール引数をそのままログに記録すると、ユーザーの検索クエリや個人情報が監査ログに残ります。上記ではハッシュのみ記録し、調査時に別途権限付きで原文を参照する設計としています。
